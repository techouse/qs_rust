9pub(super) fn decode_scalar(value: &str, charset: Charset) -> String {
10    decode_scalar_with_known_flags(value, charset, value.contains('+') || value.contains('%'))
11}

13pub(super) fn decode_scalar_with_known_flags(
14    value: &str,
15    charset: Charset,
16    has_escape_or_plus: bool,
17) -> String {
18    if !has_escape_or_plus {
19        return value.to_owned();
20    }
21
22    match charset {
23        Charset::Utf8 => decode_scalar_utf8(value),
24        Charset::Iso88591 => decode_scalar_latin1(value),
25    }
26}

28fn decode_scalar_utf8(value: &str) -> String {
29    let bytes = value.as_bytes();
30    let mut output = Vec::with_capacity(bytes.len());
31    let mut changed = false;
32    let mut index = 0usize;
33
34    while index < bytes.len() {
35        match bytes[index] {
36            b'+' => {
37                output.push(b' ');
38                changed = true;
39                index += 1;
40            }
41            b'%' if index + 2 < bytes.len() => {
42                let hi = hex_value(bytes[index + 1]);
43                let lo = hex_value(bytes[index + 2]);
44                if let (Some(hi), Some(lo)) = (hi, lo) {
45                    output.push((hi << 4) | lo);
46                    changed = true;
47                    index += 3;
48                    continue;
49                }
50                output.push(bytes[index]);
51                index += 1;
52            }
53            byte => {
54                output.push(byte);
55                index += 1;
56            }
57        }
58    }
59
60    if !changed {
61        return value.to_owned();
62    }
63
64    match String::from_utf8(output) {
65        Ok(decoded) => decoded,
66        Err(_) => replace_plus_with_space(value),
67    }
68}

70fn decode_scalar_latin1(value: &str) -> String {
71    let bytes = value.as_bytes();
72    let mut output = String::with_capacity(bytes.len());
73    let mut changed = false;
74    let mut index = 0usize;
75
76    while index < bytes.len() {
77        match bytes[index] {
78            b'+' => {
79                output.push(' ');
80                changed = true;
81                index += 1;
82            }
83            b'%' if index + 2 < bytes.len() => {
84                let hi = hex_value(bytes[index + 1]);
85                let lo = hex_value(bytes[index + 2]);
86                if let (Some(hi), Some(lo)) = (hi, lo) {
87                    output.push(char::from((hi << 4) | lo));
88                    changed = true;
89                    index += 3;
90                    continue;
91                }
92                output.push('%');
93                index += 1;
94            }
95            byte => {
96                output.push(char::from(byte));
97                index += 1;
98            }
99        }
100    }
101
102    if changed { output } else { value.to_owned() }
103}

105fn replace_plus_with_space(value: &str) -> String {
106    if value.contains('+') {
107        value.replace('+', " ")
108    } else {
109        value.to_owned()
110    }
111}

113pub(super) fn decode_component(
114    value: &str,
115    charset: Charset,
116    kind: DecodeKind,
117    options: &DecodeOptions,
118) -> String {
119    match options.decoder() {
120        Some(decoder) => decoder.decode(value, charset, kind),
121        None => decode_scalar(value, charset),
122    }
123}

125pub(super) fn interpret_numeric_entities_in_node(node: Node) -> Node {
126    match node {
127        Node::Value(Value::String(text)) => {
128            Node::scalar(Value::String(interpret_numeric_entities(&text)))
129        }
130        Node::Array(items) => Node::Array(
131            items
132                .into_iter()
133                .map(interpret_numeric_entities_in_node)
134                .collect(),
135        ),
136        other => other,
137    }
138}

140pub(super) fn interpret_numeric_entities(input: &str) -> String {
141    if !input.contains("&#") {
142        return input.to_owned();
143    }
144
145    let chars: Vec<char> = input.chars().collect();
146    let mut output = String::with_capacity(input.len());
147    let mut index = 0usize;
148
149    while index < chars.len() {
150        if chars[index] == '&' && index + 2 < chars.len() && chars[index + 1] == '#' {
151            let mut scan = index + 2;
152            let mut hex = false;
153            if scan < chars.len() && (chars[scan] == 'x' || chars[scan] == 'X') {
154                hex = true;
155                scan += 1;
156            }
157            let start_digits = scan;
158            while scan < chars.len()
159                && if hex {
160                    chars[scan].is_ascii_hexdigit()
161                } else {
162                    chars[scan].is_ascii_digit()
163                }
164            {
165                scan += 1;
166            }
167            if scan < chars.len() && chars[scan] == ';' && scan > start_digits {
168                let digits: String = chars[start_digits..scan].iter().collect();
169                let value = if hex {
170                    u32::from_str_radix(&digits, 16).ok()
171                } else {
172                    digits.parse::<u32>().ok()
173                };
174                if let Some(codepoint) = value.and_then(char::from_u32) {
175                    output.push(codepoint);
176                    index = scan + 1;
177                    continue;
178                }
179            }
180        }
181
182        output.push(chars[index]);
183        index += 1;
184    }
185
186    output
