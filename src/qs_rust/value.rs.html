5/// The object representation used throughout the crate.
6///
7/// Objects preserve insertion order to match the observable behavior of the
8/// sibling ports and the upstream `qs` ecosystem.
9pub type Object = IndexMap<String, Value>;
10
11/// A query-string-compatible value tree.
12///
13/// [`Value`] is the shared input/output representation for [`crate::encode()`]
14/// and [`crate::decode()`]. It intentionally stays close to the data model used
15/// by the sibling ports: scalars, temporal leaves, byte strings, arrays, and
16/// ordered objects.
17#[cfg_attr(feature = "serde", derive(serde::Serialize, serde::Deserialize))]
18#[derive(Clone, Debug, PartialEq)]
19pub enum Value {
20    /// A null value.
21    Null,
22    /// A boolean scalar.
23    Bool(bool),
24    /// A signed 64-bit integer scalar.
25    I64(i64),
26    /// An unsigned 64-bit integer scalar.
27    U64(u64),
28    /// A 64-bit floating-point scalar.
29    F64(f64),
30    /// A UTF-8 string scalar.
31    String(String),
32    /// A core temporal scalar.
33    Temporal(TemporalValue),
34    /// An opaque byte string scalar.
35    Bytes(Vec<u8>),
36    /// An ordered list of values.
37    Array(Vec<Value>),
38    /// An ordered object map.
39    Object(Object),
40}
41
42impl Value {
43    pub(crate) fn is_scalar(&self) -> bool {
44        matches!(
45            self,
46            Self::Null
47                | Self::Bool(_)
48                | Self::I64(_)
49                | Self::U64(_)
50                | Self::F64(_)
51                | Self::String(_)
52                | Self::Temporal(_)
53                | Self::Bytes(_)
54        )
55    }
56
57    pub(crate) fn is_empty_for_decode(&self) -> bool {
58        match self {
59            Self::Null => true,
60            Self::String(text) => text.is_empty(),
61            Self::Array(values) => values.is_empty(),
62            Self::Object(entries) => entries.is_empty(),
63            _ => false,
64        }
65    }
66}
67
68impl From<Object> for Value {
69    fn from(value: Object) -> Self {
70        Self::Object(value)
71    }
72}
