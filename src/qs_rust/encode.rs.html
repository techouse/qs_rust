27/// Encodes a [`Value`] tree into a query string.
28///
29/// The encoder preserves object insertion order by default, supports multiple
30/// list formats, and can be customized with filters, sorters, key/value
31/// encoders,
32/// and temporal serializers through [`EncodeOptions`].
33///
34/// # Errors
35///
36/// Returns [`EncodeError`] when the supplied [`EncodeOptions`] are invalid or
37/// when encoding exceeds a configured maximum depth.
38///
39/// # Examples
40///
41/// ```
42/// use qs_rust::{EncodeOptions, Object, Value, encode};
43///
44/// let mut root = Object::new();
45/// root.insert("a".to_owned(), Value::String("1".to_owned()));
46/// root.insert("b".to_owned(), Value::String("2".to_owned()));
47///
48/// let encoded = encode(&Value::Object(root), &EncodeOptions::new()).unwrap();
49/// assert_eq!(encoded, "a=1&b=2");
50/// ```
51pub fn encode(value: &Value, options: &EncodeOptions) -> Result<String, EncodeError> {
52    options.validate()?;
53
54    let root = filter_root_value(value, options);
55    let mut body = String::new();
56    let mut has_parts = false;
57
58    match root {
59        EncodeInput::Omitted => return Ok(String::new()),
60        EncodeInput::Borrowed(root) => {
61            append_root_output(root, options, &mut body, &mut has_parts)?
62        }
63        EncodeInput::Owned(root) => append_root_output(&root, options, &mut body, &mut has_parts)?,
64    }
65
66    if !has_parts || body.is_empty() {
67        return Ok(String::new());
68    }
69
70    let mut output = String::new();
71    if options.add_query_prefix {
72        output.push('?');
73    }
74
75    if options.charset_sentinel {
76        output.push_str(options.charset.sentinel());
77        output.push('&');
78    }
79
80    output.push_str(&body);
81    Ok(output)
82}

84fn append_root_output(
85    value: &Value,
86    options: &EncodeOptions,
87    body: &mut String,
88    has_parts: &mut bool,
89) -> Result<(), EncodeError> {
90    match value {
91        Value::Object(object) => {
92            for key in ordered_object_keys(object, options) {
93                let Some(child) = object.get(&key) else {
94                    continue;
95                };
96                let path =
97                    KeyPathNode::from_raw(raw_root_key_component(&key, Some(child), options));
98                append_encoded_node(body, has_parts, child, path, options, 0)?;
99            }
100        }
101        Value::Array(items) => {
102            for index in ordered_array_indices(items, options) {
103                let Some(child) = items.get(index) else {
104                    continue;
105                };
106                let path = KeyPathNode::from_raw(raw_key_component(&index.to_string(), options));
107                append_encoded_node(body, has_parts, child, path, options, 0)?;
108            }
109        }
110        _ => {}
111    }
112
113    Ok(())
114}

116fn append_encoded_node(
117    body: &mut String,
118    has_parts: &mut bool,
119    node: &Value,
120    path: KeyPathNode,
121    options: &EncodeOptions,
122    depth: usize,
123) -> Result<(), EncodeError> {
124    if let Some(fast) = try_encode_linear_map_chain(node, &path, options, depth) {
125        emit_part(body, has_parts, &options.delimiter, &fast);
126        return Ok(());
127    }
128
129    if has_filter_control(options) {
130        let encoded = encode_node_filtered(EncodeInput::Borrowed(node), path, options, depth)?;
131        append_parts(body, has_parts, &options.delimiter, encoded);
132        return Ok(());
133    }
134
135    encode_node_hot(body, has_parts, node, path, options, depth)
136}

138fn array_child_path(path: &KeyPathNode, index: usize, options: &EncodeOptions) -> KeyPathNode {
139    match options.list_format {
140        crate::options::ListFormat::Indices => path.append_bracketed_component(&index.to_string()),
141        crate::options::ListFormat::Brackets => path.append_empty_list_suffix(),
142        crate::options::ListFormat::Repeat | crate::options::ListFormat::Comma => path.clone(),
143    }
144}

146fn emit_part(body: &mut String, has_parts: &mut bool, delimiter: &str, part: &str) {
147    if part.is_empty() {
148        return;
149    }
150    if *has_parts {
151        body.push_str(delimiter);
152    }
153    body.push_str(part);
154    *has_parts = true;
155}
156
157fn append_parts(body: &mut String, has_parts: &mut bool, delimiter: &str, parts: Vec<String>) {
158    for part in parts {
159        emit_part(body, has_parts, delimiter, &part);
160    }
161}

163fn ordered_object_keys(object: &Object, options: &EncodeOptions) -> Vec<String> {
164    let mut keys = match options.filter.as_ref() {
165        Some(crate::options::EncodeFilter::Whitelist(whitelist)) => whitelist
166            .iter()
167            .filter_map(|selector| match selector {
168                WhitelistSelector::Key(key) => Some(key.clone()),
169                WhitelistSelector::Index(_) => None,
170            })
171            .collect::<Vec<_>>(),
172        _ => object.keys().cloned().collect::<Vec<_>>(),
173    };
174
175    if let Some(sorter) = options.sorter.as_ref() {
176        keys.sort_by(|left, right| sorter.compare(left, right));
177    } else if matches!(options.sort, SortMode::LexicographicAsc) {
178        keys.sort();
179    }
180
181    keys
182}
183
184fn ordered_array_indices(items: &[Value], options: &EncodeOptions) -> Vec<usize> {
185    match options.filter.as_ref() {
186        Some(crate::options::EncodeFilter::Whitelist(whitelist)) => whitelist
187            .iter()
188            .filter_map(|selector| match selector {
189                WhitelistSelector::Index(index) => Some(*index),
190                WhitelistSelector::Key(_) => None,
191            })
192            .collect(),
193        _ => (0..items.len()).collect(),
194    }
195}

197fn raw_key_component(key: &str, options: &EncodeOptions) -> String {
198    if options.allow_dots && options.encode_dot_in_keys {
199        key.replace('.', encoded_dot_escape(options))
200    } else {
201        key.to_owned()
202    }
203}
204
205fn raw_root_key_component(key: &str, value: Option<&Value>, options: &EncodeOptions) -> String {
206    if options.allow_dots
207        && options.encode_dot_in_keys
208        && matches!(value, Some(Value::Array(_)) | Some(Value::Object(_)))
209    {
210        key.replace('.', encoded_dot_escape(options))
211    } else {
212        key.to_owned()
213    }
214}
