1//! The encode linear-chain fast path.
2
3use crate::key_path::KeyPathNode;
4use crate::options::{EncodeOptions, ListFormat, SortMode};
5use crate::value::Value;
6
7use super::raw_key_component;
8use super::scalar::encode_scalar_leaf;
9
10pub(super) fn try_encode_linear_map_chain(
11    value: &Value,
12    path: &KeyPathNode,
13    options: &EncodeOptions,
14    depth: usize,
15) -> Option<String> {
16    if options.strict_null_handling
17        || options.skip_nulls
18        || options.allow_empty_lists
19        || options.encode_values_only
20        || options.allow_dots
21        || options.encode_dot_in_keys
22        || options.filter.is_some()
23        || options.sorter.is_some()
24        || options.encoder.is_some()
25        || options.has_temporal_serializer()
26        || !matches!(options.list_format, ListFormat::Indices)
27        || options.comma_round_trip
28        || options.comma_compact_nulls
29        || !matches!(options.sort, SortMode::Preserve)
30        || options.max_depth.is_some_and(|max| depth > max)
31    {
32        return None;
33    }
34
35    let mut current = value;
36    let mut materialized = path.materialize().to_owned();
37    let mut current_depth = depth;
38
39    loop {
40        match current {
41            Value::Object(object) => {
42                if object.len() != 1 {
43                    return None;
44                }
45                current_depth = current_depth.saturating_add(1);
46                if options.max_depth.is_some_and(|max| current_depth > max) {
47                    return None;
48                }
49                let (key, next) = object.iter().next()?;
50                append_bracketed_path_component(
51                    &mut materialized,
52                    &raw_key_component(key, options),
53                );
54                current = next;
55            }
56            Value::Array(_) => return None,
57            scalar => return encode_scalar_leaf(scalar, &materialized, options),
58        }
59    }
60}
61
62fn append_bracketed_path_component(path: &mut String, component: &str) {
63    path.push('[');
64    path.push_str(component);
65    path.push(']');
66}
