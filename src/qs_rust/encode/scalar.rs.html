7pub(super) fn encode_scalar_leaf(
8    node: &Value,
9    key: &str,
10    options: &EncodeOptions,
11) -> Option<String> {
12    if scalar_is_null_like(node, options) {
13        if options.skip_nulls {
14            return None;
15        }
16        if options.strict_null_handling {
17            return Some(finalize_key_only_fragment(key, options));
18        }
19        let key = finalize_key_path(key, options);
20        return Some(format!("{key}="));
21    }
22
23    let key = finalize_key_path(key, options);
24    let value = encode_string_or_raw(
25        &encoded_scalar_text(node, options).expect("non-null scalar leaf should resolve"),
26        options,
27    );
28    Some(format!("{key}={value}"))
29}

31pub(super) fn finalize_key_path(key: &str, options: &EncodeOptions) -> String {
32    let key = key_token_text(key, options);
33    format_key_text(&key, options)
34}

36pub(super) fn finalize_key_only_fragment(key: &str, options: &EncodeOptions) -> String {

44pub(super) fn joined_text_value(text: &str, options: &EncodeOptions) -> String {
45    if let Some(encoder) = options.encoder.as_ref() {
46        return encoder.encode(

56pub(super) fn encode_string_or_raw(text: &str, options: &EncodeOptions) -> String {
57    if options.encode {
58        encode_with_charset(text, options.charset, options.format)
59    } else {
60        text.to_owned()
61    }

64pub(super) fn encode_key_only_fragment(key: &str, options: &EncodeOptions) -> String {
65    let formatted = format_key_text(key, options);
66    if options.encode && matches!(options.format, Format::Rfc1738) {
67        formatted.replace('+', "%20")
68    } else {
69        formatted
70    }
71}

73pub(super) fn format_key_text(text: &str, options: &EncodeOptions) -> String {
74    if options.encode && !options.encode_values_only {
75        encode_with_charset(text, options.charset, options.format)
76    } else {
77        text.to_owned()
78    }
79}

81pub(super) fn encode_with_charset(text: &str, charset: Charset, format: Format) -> String {
82    match charset {
83        Charset::Utf8 => percent_encode_bytes(text.as_bytes(), format),
84        Charset::Iso88591 => percent_encode_latin1(text, format),
85    }
86}
87
88pub(super) fn percent_encode_latin1(text: &str, format: Format) -> String {
89    let mut output = String::new();
90    for ch in text.chars() {
91        let code = ch as u32;
92        if let Ok(byte) = u8::try_from(code) {
93            append_encoded_byte(byte, format, &mut output);
94        } else {
95            let numeric = format!("&#{code};");
96            output.push_str(&percent_encode_bytes(numeric.as_bytes(), Format::Rfc3986));
97        }
98    }
99    output
100}

102pub(super) fn percent_encode_bytes(bytes: &[u8], format: Format) -> String {
103    let mut output = String::with_capacity(bytes.len() * 3);
104    for &byte in bytes {
105        append_encoded_byte(byte, format, &mut output);
106    }
107    output
108}
109
110fn append_encoded_byte(byte: u8, format: Format, output: &mut String) {
111    let is_alphanumeric = byte.is_ascii_alphanumeric();
112    let is_safe = matches!(byte, b'-' | b'.' | b'_' | b'~');
113    if is_alphanumeric
114        || is_safe
115        || (matches!(format, Format::Rfc1738) && matches!(byte, b'(' | b')'))
116    {
117        output.push(byte as char);
118        return;
119    }
120
121    if matches!(format, Format::Rfc1738) && byte == b' ' {
122        output.push('+');
123        return;
124    }
125
126    output.push('%');
127    output.push(HEX[(byte >> 4) as usize] as char);
128    output.push(HEX[(byte & 0x0F) as usize] as char);
129}

131fn decode_bytes(bytes: &[u8], charset: Charset) -> String {
132    match charset {
133        Charset::Utf8 => String::from_utf8_lossy(bytes).into_owned(),
134        Charset::Iso88591 => bytes.iter().map(|byte| char::from(*byte)).collect(),
135    }
136}

138pub(super) fn plain_string_for_comma(value: &Value, options: &EncodeOptions) -> String {
139    match value {
140        Value::Array(values) => values
141            .iter()
142            .map(|value| plain_string_for_comma(value, options))
143            .collect::<Vec<_>>()
144            .join(","),
145        Value::Object(_) => "[object Object]".to_owned(),
146        _ => plain_scalar_text(value, options).unwrap_or_default(),
147    }
148}

150pub(super) fn scalar_is_null_like(value: &Value, options: &EncodeOptions) -> bool {
151    match value {
152        Value::Null => true,
153        Value::Temporal(temporal) => plain_temporal_text(temporal, options).is_none(),
154        _ => false,
155    }
156}

158pub(super) fn encoded_scalar_text(value: &Value, options: &EncodeOptions) -> Option<String> {
159    match value {
160        Value::Null => None,
161        Value::Temporal(temporal) => {
162            let text = plain_temporal_text(temporal, options)?;
163            Some(encoded_text_token(&text, options))
164        }
165        Value::Array(values) => values
166            .iter()
167            .map(|value| plain_string_for_comma(value, options))
168            .collect::<Vec<_>>()
169            .join(",")
170            .into(),
171        Value::Object(_) => Some("[object Object]".to_owned()),
172        _ => Some(value_token_text(value, options)),
173    }
174}

182fn key_token_text(key: &str, options: &EncodeOptions) -> String {
183    if options.encode_values_only {
184        return key.to_owned();
185    }
186
187    if let Some(encoder) = options.encoder.as_ref() {
188        return encoder.encode(EncodeToken::Key(key), options.charset, options.format);
189    }
190
191    key.to_owned()
192}

194fn plain_scalar_text(value: &Value, options: &EncodeOptions) -> Option<String> {
195    match value {
196        Value::Null => None,
197        Value::Bool(boolean) => Some(boolean.to_string()),
198        Value::I64(number) => Some(number.to_string()),
199        Value::U64(number) => Some(number.to_string()),
200        Value::F64(number) => Some(number.to_string()),
201        Value::String(text) => Some(text.clone()),
202        Value::Temporal(temporal) => plain_temporal_text(temporal, options),
203        Value::Bytes(bytes) => Some(decode_bytes(bytes, options.charset)),
204        Value::Array(values) => Some(
205            values
206                .iter()
207                .map(|value| plain_string_for_comma(value, options))
208                .collect::<Vec<_>>()
209                .join(","),
210        ),
211        Value::Object(_) => Some("[object Object]".to_owned()),
212    }
213}

215fn plain_temporal_text(value: &TemporalValue, options: &EncodeOptions) -> Option<String> {
216    options.temporal_serializer().map_or_else(
217        || Some(value.to_string()),
218        |serializer| serializer.serialize(value),
219    )
220}

222fn encoded_text_token(text: &str, options: &EncodeOptions) -> String {
223    if let Some(encoder) = options.encoder.as_ref() {
224        return encoder.encode(
225            EncodeToken::TextValue(text),
226            options.charset,
227            options.format,
228        );
229    }

231    text.to_owned()
232}

234fn value_token_text(value: &Value, options: &EncodeOptions) -> String {
235    if let Some(encoder) = options.encoder.as_ref() {
236        return encoder.encode(EncodeToken::Value(value), options.charset, options.format);
