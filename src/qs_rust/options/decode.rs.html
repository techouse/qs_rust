8/// Options that control query-string decoding.
9///
10/// The defaults are chosen to match the common `qs` behavior: UTF-8 decoding,
11/// `&` as the delimiter, five levels of depth, list parsing enabled, and
12/// duplicate values combined into arrays.
13#[derive(Clone, Debug)]
14pub struct DecodeOptions {
15    pub(crate) allow_dots: bool,
16    pub(crate) decode_dot_in_keys: bool,
17    pub(crate) allow_empty_lists: bool,
18    pub(crate) allow_sparse_lists: bool,
19    pub(crate) list_limit: usize,
20    pub(crate) charset: Charset,
21    pub(crate) charset_sentinel: bool,
22    pub(crate) comma: bool,
23    pub(crate) delimiter: Delimiter,
24    pub(crate) depth: usize,
25    pub(crate) parameter_limit: usize,
26    pub(crate) duplicates: Duplicates,
27    pub(crate) ignore_query_prefix: bool,
28    pub(crate) interpret_numeric_entities: bool,
29    pub(crate) parse_lists: bool,
30    pub(crate) strict_depth: bool,
31    pub(crate) strict_null_handling: bool,
32    pub(crate) throw_on_limit_exceeded: bool,
33    pub(crate) decoder: Option<DecodeDecoder>,
34}
35
36impl Default for DecodeOptions {
37    fn default() -> Self {
38        Self {
39            allow_dots: false,
40            decode_dot_in_keys: false,
41            allow_empty_lists: false,
42            allow_sparse_lists: false,
43            list_limit: 20,
44            charset: Charset::Utf8,
45            charset_sentinel: false,
46            comma: false,
47            delimiter: Delimiter::default(),
48            depth: 5,
49            parameter_limit: 1000,
50            duplicates: Duplicates::Combine,
51            ignore_query_prefix: false,
52            interpret_numeric_entities: false,
53            parse_lists: true,
54            strict_depth: false,
55            strict_null_handling: false,
56            throw_on_limit_exceeded: false,
57            decoder: None,
58        }
59    }
60}

292    pub(crate) fn validate(&self) -> Result<(), DecodeError> {
293        if self.parameter_limit == 0 {
294            return Err(DecodeError::InvalidParameterLimit);
295        }
296
297        if self.decode_dot_in_keys && !self.allow_dots {
298            return Err(DecodeError::DecodeDotInKeysRequiresAllowDots);
299        }
300
301        if matches!(&self.delimiter, Delimiter::String(text) if text.is_empty()) {
302            return Err(DecodeError::EmptyDelimiter);
303        }
304
305        Ok(())
306    }
